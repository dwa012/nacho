def nacho_select(method, choices = nil, options = {}, html_options = {}, &block)
  nacho_options = Nacho::Common::Lib.build_options(method, choices, options, html_options)
  select_element = select(method, nacho_options[:choices], nacho_options[:options], nacho_options[:html_options], *block)
  select_element += nacho_options[:button] if nacho_options[:html_options][:multiple]

  select_element
end
